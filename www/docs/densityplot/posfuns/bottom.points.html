directlabels - densityplot - Positioning Method - bottom.points

Positioning Method for the bottom of a group of points.

bottom.points <-
  gapply.fun(transform(d[which.min(d$y),],hjust=0.5,vjust=1))

chemscore

data(Chem97,package="mlmRev")
library(lattice)
p <- densityplot(~gcsescore|gender,Chem97,
            groups=factor(score),layout=c(1,2),
            n=500,plot.points=FALSE)
direct.label(p,"bottom.points")
