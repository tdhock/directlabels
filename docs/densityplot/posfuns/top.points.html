directlabels - densityplot - Positioning Method - top.points

Positioning Method for the top of a group of points.

top.points <-
  gapply.fun(transform(d[which.max(d$y),],hjust=0.5,vjust=0))

chemscore

data(Chem97,package="mlmRev")
library(lattice)
p <- densityplot(~gcsescore|gender,Chem97,
            groups=factor(score),layout=c(1,2),
            n=500,plot.points=FALSE)
direct.label(p,"top.points")
